Module type Round.RoundSig

module type RoundSig = sig .. end
type t

t is the type of a round.

type difficulty =
| Easy
| Medium
| Hard
| Invalid

difficulty are the possible levels of difficulty for a round.

type result =
| Valid of t
| Invalid of string

result is the result of making a move on a round. Either valid or invalid.

val new_round : difficulty -> t

new_round d is a new game of difficulty d.

val deal : t -> result
val play : Card.card -> t -> result
val pass : Card.card list -> t -> result
val hand : t -> Partialdeck.PartialDeck.t

hand t is the players hand at the current state.

val pile : t -> (Card.card * int) list

pile t is the pile at the current state.

val next : t -> t

next t is t with the state incremented.

val description : t -> string

description t is the description for the current state.

val is_next : t -> bool

is_next t is true if there is a next state, otherwise false.

val bot_hand : t -> int -> Partialdeck.PartialDeck.t

bot_hand t id is bot id's current hand. Meant only for use with bots.

val bot_pile : t -> (Card.card * int) list

bot_pile t is the current pile. Meant only for use with bots.

val round_score : t -> int list

round_score t is a list of all the players round score. In order of their id.

val total_score : t -> int list

total_score t is a list of all the players cumulative score. In order of their id.

val names : t -> string list

names t is the names of the players in id order.

val string_of_round : t -> string
val next_action : t -> string

next_action is the string of the next action.

val test_deal : Partialdeck.PartialDeck.t list -> t

test_deal pdl is a new round where pdl are dealt as each hand. The length of pdl must be 4.
