Module Partialdeck.PartialDeck

module PartialDeck: PartialDeckSig 
type t

t is the type of a PartialDeck.

val empty : t

empty is an empty partial deck.

val full : t

full is a partial deck filled with all the cards in a standard deck.

val insert : Card.card -> t -> t

insert card t is deck t with card inserted. Raises DuplicateCard if card was already in the deck.

val remove : Card.card -> t -> t

remove card t is a partial deck t without card. Raises CardNotFound if card is not in t.

val move : Card.card ->
       t ->
       t ->
       t * t

move card t1 t2 is a tuple of (t1,t2) where card is moved from t1 to t2. Raises CardNotFound if card is not in t1. Raises DuplicateCare if card is already in t2.

val move_at_index : int ->
       t ->
       t ->
       t * t

move i t1 t2 is a tuple of (t1,t2) where the card at index i is moved from t1 to t2. Raises CardNotFound if card is not in t1. Raises DuplicateCare if card is already in t2.

val mem : Card.card -> t -> bool

mem card t is true if card is in t, otherwise false.

val size : t -> int

size t is the number of cards in t.

val is_empty : t -> bool

is_empty t is true if t contains zero cards. Otherwise false.

val random_card : t -> Card.card option

random_card t is a random card option in t. None if t is empty.

val to_list : t -> (Card.card * int) list

to_list t is a list of the cards in t in a tuple with their index.

val find : int -> t -> Card.card option

find i t is the ith Some card, if the ith card does not exist, None.

val voided : Card.suite -> t -> bool

voided suite t is true if there are none of suite in t.

val count_points : t -> int

count_points t is the number of points in partial deck t.

val merge : t ->
       t -> t

merge t1 t2 is the partial deck containing all the cards in t1 and t2.

val contains_hearts : t -> bool

contains_hearts t is true if t contains any hearts. Otherwise false.

val string_of_partialdeck : t -> string

string_of_partialdeck t is a string representing the deck t.

val lowest : t -> Card.suite -> Card.card

lowest t suite is the lowest card in t of suite.

val highest : t -> Card.suite -> Card.card

highest t suite is the highest card in t of suite.

val remove_cards : Card.card list ->
       t -> t

remove_cards c_l t is t without any cards from c_l.

val add_cards : Card.card list ->
       t -> t

add_cards c_l t is t with all the cards from c_l.

val shoot_the_moon : t -> bool

shoot_the_moon t is true if t has all the cards needed to shoot the moon. Otherwise false.

val only_bad : t -> bool

only_bad t is true if t contains only cards that give points. Otherwise false.
